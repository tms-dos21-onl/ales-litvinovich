Конфигурация разбита на две части: фронт и бэкенд
```bash
server {
    listen 80 default_server;
    listen [::]:80 default_server;

    root /var/www/build;
    index index.html index.htm index.nginx-debian.html;

    server_name site1.com;

    location / {
        try_files $uri $uri/ /index.html;
    }
}

server {
    listen 81 default_server;
    listen [::]:81 default_server;

    server_name site1.com;
 
     location / {
```
